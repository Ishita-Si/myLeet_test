class Solution {
public:
    string processStr(string s) {
        string result = "";

        for (char x : s) {
            if (x == '*'){
                if (!result.empty()) {
                    result.pop_back();
                }
            }
            else if (x == '#')
                result.insert(result.end(), result.begin(), result.end());
            else if (x == '%')
                reverse(result.begin(), result.end());
            else {
                result.push_back(x);
            }
        }
        return result;
    }
};
