class Solution {
public:
    string mapWordWeights(vector<string>& words, vector<int>& weights) {
        string t;
        int i = 0;
        for(string x : words){
            int count = 0;
            for(char y : x){
                count += weights[y-'a'];
            }
            char z = 'a' + 25 - count%26;

            t.push_back(z);
        }
        return t;
    }
};
