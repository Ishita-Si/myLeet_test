class Solution {
public:
    long long int robber(vector<int>& nums) {
        
        int n = nums.size();
        int prev = nums[0];
        int prev2 = 0;

        for(int idx = 1 ; idx < nums.size(); idx++){
            int house = nums[idx];
            if(idx > 1) house +=  prev2;
            int notHouse = prev;

            int curr = max(house, notHouse);
            prev2 = prev;
            prev = curr;
        }
        return prev;
    }
    long long int rob(vector<int>& nums) {
        int n = nums.size();
        vector<int> temp1, temp2;
        if(n == 1) return nums[0];

        for(int i = 0; i < n; i++){
            if(i != 0) temp1.push_back(nums[i]);
            if(i != n-1) temp2.push_back(nums[i]);
        }
       return max(robber(temp1), robber(temp2)); 
    }
};
