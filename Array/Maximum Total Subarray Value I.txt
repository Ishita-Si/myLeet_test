class Solution {
public:
    long long maxTotalValue(vector<int>& nums, int k) {
        int minElm = INT_MAX;
        int maxElm = 0;
        for(int x : nums){
            maxElm = max(x,maxElm);
            minElm = min(x,minElm); 
        }

        return (long long)k * (maxElm - minElm);
    }
};
