class Solution {
public:
    int getCommon(vector<int>& nums1, vector<int>& nums2) {

    unordered_set<int> st;
    for(int x : nums1){
        st.insert(x);
    }

    int minCom = INT_MAX;

    for(int x : nums2){
        if(st.count(x)) 
            minCom = min(minCom,x);
    }
        return minCom == INT_MAX?-1:minCom;
    }
};
