class Solution {
public:
    double angleClock(int hour, int minutes) {
        double hr = ((hour % 12)* 30) + (minutes * 0.5);
        double mins = (minutes % 60)*6;

        return min(abs(hr-mins), 360 - abs(hr-mins));
    }
};
